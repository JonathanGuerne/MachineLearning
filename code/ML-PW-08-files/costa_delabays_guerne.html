def disto_fun(cs,X, preds):
    sum_arr = []
    for n in range(X.shape[0]):
        id = int(preds[n])
        x=X[n]
        c = cs[id]
        sum_arr.append(np.linalg.norm(x-c))
    return np.sum(sum_arr)
        

def k_mean(k, X, iter_max=20, epsilon=0):

    prev_centers = np.zeros((X.shape[0],k))
    centers = X[np.random.randint(0,X.shape[0],k)]

    iter = 0
    
    preds = np.zeros(X.shape[0])
    
    costs = []
    converged = False
    
    while iter < iter_max and not converged:
        
        prev_centers = centers.copy()
        for i in range(X.shape[0]):
            # closest center cc
            x = X[i]
            preds[i] = np.argmin([np.linalg.norm(x-c) for c in centers])
        
        for j in range(k):
            centers[j] = np.mean(X[np.argwhere(preds == j)], axis=0)
            
        iter += 1
        costs.append(disto_fun(centers,X,preds))
        
        converged = np.array_equal(prev_centers,centers)
        
    preds = np.zeros(X.shape[0])
    for i in range(X.shape[0]):
        # closest center cc
        x = X[i]
        preds[i] = np.argmin([np.linalg.norm(x-c) for c in centers])

    return preds, centers, costs

def run_k_mean(k_range,iteration_per_k,X):
    svd_preds = None
    svd_centers = None
    svd_costs = None

    for k in k_range:
        for i in range(iteration_per_k):
            preds, centers, costs = k_mean(k,X)
            if svd_costs is None or svd_costs[-1] > costs[-1]:
                svd_preds = preds
                svd_centers = centers
                svd_costs = costs
                
    return svd_preds,svd_centers,svd_costs

  • le nombre d'itération nécessaire pour atteindre la convergence dépend de la position initiale des centroïdes.
  • les positions initiales des centroïdes sont aléatoirement sélectionner parmis l'ensemble de point
  • pour contrer les problèmes récurrent de minimum locaux on lance plusieurs fois l'algorithme k-means en prenant le résultats qui donne obtient la valeur de coût la plus basse
  • dans notre implémentation la convergence est atteinte si la valeur de coût entre 2 itérations de k-mean est identique .i.e si rien n'a bougé téhoriquement. Cette approche fonctionne dans ce problème mais pourrait utiliser un epsilon pour permettre une certaine marge.
